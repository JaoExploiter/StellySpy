			tweenBack:Play()
		end)
	end
	
	changeColorToRedAndBack()
	
end;

-- StarterGui.Stop The Cap.Minimized.Bar.SmoothDrag
local function C_a()
local script = G2L["a"];
	-- Need help? Check me out on the devforum!
	-- Link: https://devforum.roblox.com/t/smooth-dragging-easily-drag-your-frames/2508276
	
	local Drag = script.Parent
	gsCoreGui = game:GetService("CoreGui")
	gsTween = game:GetService("TweenService")
	local UserInputService = game:GetService("UserInputService")
		local dragging
		local dragInput
		local dragStart
		local startPos
		local function update(input)
			local delta = input.Position - dragStart
			local dragTime = 0.04
			local SmoothDrag = {}
			SmoothDrag.Position = UDim2.new(startPos.X.Scale, startPos.X.Offset + delta.X, startPos.Y.Scale, startPos.Y.Offset + delta.Y)
			local dragSmoothFunction = gsTween:Create(Drag, TweenInfo.new(dragTime, Enum.EasingStyle.Sine, Enum.EasingDirection.InOut), SmoothDrag)
			dragSmoothFunction:Play()
		end
		Drag.InputBegan:Connect(function(input)
			if input.UserInputType == Enum.UserInputType.MouseButton1 or input.UserInputType == Enum.UserInputType.Touch then
				dragging = true
				dragStart = input.Position
				startPos = Drag.Position
				input.Changed:Connect(function()
					if input.UserInputState == Enum.UserInputState.End then
						dragging = false
					end
				end)
			end
		end)
		Drag.InputChanged:Connect(function(input)
			if input.UserInputType == Enum.UserInputType.MouseMovement or input.UserInputType == Enum.UserInputType.Touch then
				dragInput = input
			end
		end)
		UserInputService.InputChanged:Connect(function(input)
			if input == dragInput and dragging and Drag.Size then
				update(input)
			end
		end)
	
end;
task.spawn(C_a);
-- StarterGui.Stop The Cap.LocalScript
local function C_b()
local script = G2L["b"];
	local ReplicatedStorage = game:GetService("ReplicatedStorage")
	local executedRemotes = {}
	
	local textbox = script.Parent.MainFrame.TextBox
	local buttons = script.Parent.MainFrame.Remotes
	local button1 = buttons.Remote1
	local button2 = buttons.Remote2
	local button3 = buttons.Remote3
	local button4 = buttons.Remote4
	local button5 = buttons.Remote5
	local button6 = buttons.Remote6
	local button7 = buttons.Remote7
	local button8 = buttons.Remote8
	local button9 = buttons.Remote9
	local button10 = buttons.Remote10
	local button11 = buttons.Remote11
	local button12 = buttons.Remote12
	local button13 = buttons.Remote13
	local button14 = buttons.Remote14
	local runButton = script.Parent.MainFrame.MainButtons.Run
	local dup = script.Parent.MainFrame.MainButtons.Dup
	local ExcludeLogs = script.Parent.MainFrame.MainButtons.ExcludeLogs
	local Excludermt = script.Parent.MainFrame.MainButtons.ExcludeRmt
	
	local buttonList = {button1, button2, button3, button4, button5, button6, button7, button8, button9, button10, button11, button12, button13, button14}
	local selectedRemote = nil
	
	
	local function generateCode(remote)
		if remote:IsA("RemoteEvent") then
			return string.format("game:GetService('ReplicatedStorage').%s:FireServer(arg1, arg2)", remote.Name)
		elseif remote:IsA("RemoteFunction") then
			return string.format("local RemoteFunction = game:GetService('ReplicatedStorage').%s:InvokeServer(arg1, arg2)", remote.Name)
		end
	end
	
	local function assignButton(remote, button)
		
		button.Text = remote.Name
		button.Visible = true
		button.MouseButton1Click:Connect(function()
			selectedRemote = remote
			textbox.Text = generateCode(remote)
			
		end)
	end
	
	local function processExecutedRemote(remote)
		if executedRemotes[remote] then return end
		executedRemotes[remote] = true
	
		for _, button in ipairs(buttonList) do
			if not button.Visible then
				assignButton(remote, button)
				return
			end
		end
	
		assignButton(remote, button1)
	end
	
	local function monitorRemotes()
		for _, remote in ipairs(ReplicatedStorage:GetDescendants()) do
			if remote:IsA("RemoteEvent") or remote:IsA("RemoteFunction") then
				processExecutedRemote(remote)
			end
		end
	
		ReplicatedStorage.DescendantAdded:Connect(function(descendant)
			if descendant:IsA("RemoteEvent") or descendant:IsA("RemoteFunction") then
				processExecutedRemote(descendant)
			end
		end)
	end
	
	monitorRemotes()
	
	runButton.MouseButton1Click:Connect(function()
		if selectedRemote then
			if selectedRemote:IsA("RemoteEvent") then
				selectedRemote:FireServer()
			elseif selectedRemote:IsA("RemoteFunction") then
				selectedRemote:InvokeServer()
			end
			print(selectedRemote.Name .. " Fired!")
		else
			textbox.Text = "No remotes selected!"
		end
	end)
	
	
	dup.MouseButton1Click:Connect(function()
		if selectedRemote then
			local clonedRemote = selectedRemote:Clone()
			clonedRemote.Parent = ReplicatedStorage
		end
	end)
	
	ExcludeLogs.MouseButton1Click:Connect(function()
		if selectedRemote then
			for _, button in ipairs(buttonList) do
				if button.Text == selectedRemote.Name then
					button.Text = ""
					button.Visible = false
					break
				end
			end
		end
	end)
	
	
	Excludermt.MouseButton1Click:Connect(function()
		
		textbox.Text = "Deleting..."
		wait(4)
		selectedRemote:Destroy()
		textbox.Text = "Remote Deleted!(he dont dissapear use ExcludeLog)"
		
	end)
	
	if true then
		
		warn("{------------------------------}")
		warn("     Made only by ._.kinzin     ")
		warn("{------------------------------}")
	
	end
end;
task.spawn(C_b);
